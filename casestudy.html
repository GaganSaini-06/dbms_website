Question 1: TOP THREE STUDENTS

Conditions:
- No subject is "A" (not absent)
- No subject has marks < 33 (not fail)
- Total % > 60% 


  QUERY:-   SELECT Rollno, 
            name, 
            dbms, 
            os, 
            spm, 
            punjabi, 
            (CAST(dbms AS SIGNED) + CAST(os AS SIGNED) + CAST(spm AS SIGNED) + CAST(punjabi AS SIGNED)) AS total_marks, 
            ROUND((CAST(dbms AS SIGNED) + CAST(os AS SIGNED) + CAST(spm AS SIGNED) + CAST(punjabi AS SIGNED)) / 4, 2) AS percentage
            FROM studentmarks
            WHERE 
            dbms NOT IN ('A', '<33') AND os NOT IN ('A', '<33') 
            AND spm NOT IN ('A', '<33') AND punjabi NOT IN ('A', '<33')
            AND CAST(dbms AS SIGNED) > 60 AND CAST(os AS SIGNED) > 60 
            AND CAST(spm AS SIGNED) > 60 AND CAST(punjabi AS SIGNED) > 60
            ORDER BY total_marks DESC
            LIMIT 3;


OUTPUT:-    +--------+--------+------+------+------+---------+-------------+------------+
            | Rollno | name   | dbms | os   | spm  | punjabi | total_marks | percentage |

Question 2: TOP THREE STUDENTS ON ALL SUBJECTS

Conditions:
- Total % > 60%
- If marks tie, then we will check AGGREGATE MARKS and if the AGGREGATE MARKS tie then the prize will given to all students for that particular position. 


  QUERY:-   -- DBMS TOP 3
  SELECT * FROM (
      SELECT 
          RollNo, name, 'DBMS' AS Subject,
          CAST(dbms AS UNSIGNED) AS Marks,
          (CAST(dbms AS UNSIGNED) + CAST(os AS UNSIGNED) + CAST(spm AS UNSIGNED) + CAST(punjabi AS UNSIGNED)) AS Total,
          ROUND((
              (CAST(dbms AS UNSIGNED) + CAST(os AS UNSIGNED) + CAST(spm AS UNSIGNED) + CAST(punjabi AS UNSIGNED)) / 4
          ), 2) AS Percentage,
          RANK() OVER (
              ORDER BY CAST(dbms AS UNSIGNED) DESC,
                       (CAST(dbms AS UNSIGNED) + CAST(os AS UNSIGNED) + CAST(spm AS UNSIGNED) + CAST(punjabi AS UNSIGNED)) DESC
          ) AS pos
      FROM StudentMarks
      WHERE dbms != 'A' AND CAST(dbms AS UNSIGNED) >= 33
  ) AS dbms_rank
  WHERE pos <= 3
  
  UNION ALL
  
  -- OS TOP 3
  SELECT * FROM (
      SELECT 
          RollNo, name, 'OS' AS Subject,
          CAST(os AS UNSIGNED) AS Marks,
          (CAST(dbms AS UNSIGNED) + CAST(os AS UNSIGNED) + CAST(spm AS UNSIGNED) + CAST(punjabi AS UNSIGNED)) AS Total,
          ROUND((
              (CAST(dbms AS UNSIGNED) + CAST(os AS UNSIGNED) + CAST(spm AS UNSIGNED) + CAST(punjabi AS UNSIGNED)) / 4
          ), 2) AS Percentage,
          RANK() OVER (
              ORDER BY CAST(os AS UNSIGNED) DESC,
                       (CAST(dbms AS UNSIGNED) + CAST(os AS UNSIGNED) + CAST(spm AS UNSIGNED) + CAST(punjabi AS UNSIGNED)) DESC
          ) AS pos
      FROM StudentMarks
      WHERE os != 'A' AND CAST(os AS UNSIGNED) >= 33
  ) AS os_rank
  WHERE pos <= 3
  
  UNION ALL
  
  -- SPM TOP 3
  SELECT * FROM (
      SELECT 
          RollNo, name, 'SPM' AS Subject,
          CAST(spm AS UNSIGNED) AS Marks,
          (CAST(dbms AS UNSIGNED) + CAST(os AS UNSIGNED) + CAST(spm AS UNSIGNED) + CAST(punjabi AS UNSIGNED)) AS Total,
          ROUND((
              (CAST(dbms AS UNSIGNED) + CAST(os AS UNSIGNED) + CAST(spm AS UNSIGNED) + CAST(punjabi AS UNSIGNED)) / 4
          ), 2) AS Percentage,
          RANK() OVER (
              ORDER BY CAST(spm AS UNSIGNED) DESC,
                       (CAST(dbms AS UNSIGNED) + CAST(os AS UNSIGNED) + CAST(spm AS UNSIGNED) + CAST(punjabi AS UNSIGNED)) DESC
          ) AS pos
      FROM StudentMarks
      WHERE spm != 'A' AND CAST(spm AS UNSIGNED) >= 33
  ) AS spm_rank
  WHERE pos <= 3
  
  UNION ALL
  
  -- PUNJABI TOP 3
  SELECT * FROM (
      SELECT 
          RollNo, name, 'Punjabi' AS Subject,
          CAST(punjabi AS UNSIGNED) AS Marks,
          (CAST(dbms AS UNSIGNED) + CAST(os AS UNSIGNED) + CAST(spm AS UNSIGNED) + CAST(punjabi AS UNSIGNED)) AS Total,
          ROUND((
              (CAST(dbms AS UNSIGNED) + CAST(os AS UNSIGNED) + CAST(spm AS UNSIGNED) + CAST(punjabi AS UNSIGNED)) / 4
          ), 2) AS Percentage,
          RANK() OVER (
              ORDER BY CAST(punjabi AS UNSIGNED) DESC,
                       (CAST(dbms AS UNSIGNED) + CAST(os AS UNSIGNED) + CAST(spm AS UNSIGNED) + CAST(punjabi AS UNSIGNED)) DESC
          ) AS pos
      FROM StudentMarks
      WHERE punjabi != 'A' AND CAST(punjabi AS UNSIGNED) >= 33
  ) AS punjabi_rank
  WHERE pos <= 3;
